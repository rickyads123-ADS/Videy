// Realizar una solicitud AJAX al servidor para obtener el contador actualizado function obtenerContador() { var xhr = new XMLHttpRequest(); xhr.onreadystatechange = function() { if (xhr.readyState === XMLHttpRequest.DONE) { if (xhr.status === 200) { var contador = parseInt(xhr.responseText); actualizarContador(contador); } } }; xhr.open("GET", "contador.txt", true); xhr.send(); } // Actualizar el contador en la página function actualizarContador(contador) { var contadorSpan = document.getElementById("contador"); contadorSpan.textContent = contador; } // Obtener el contador cuando la página se carga por primera vez window.onload = function() { obtenerContador(); // Actualizar el contador cada segundo setInterval(obtenerContador, 1000); };
